API Reference

Endpoints

Currently available endpoint for searching and retrieving download links.

GET /api/download

Search for a movie or series and retrieve download links with full metadata. Supports quality selection, series episode targeting, and subtitle retrieval.

Query Parameters
Parameter Type Status Description
querystring required Movie or series name to search
typestring optional Default: auto-detect movie or series
seasoninteger conditional Season number — required when type is series
episodeinteger conditional Episode number — required when type is series
qualitystring optional Default: best Preferred quality: 360 · 480 · 720 · 1080

Reference

Response Codes

Standard HTTP response codes returned by the Zentrix API.

200
OK
Request successful. Download links and metadata returned.
400
Bad Request
Missing required parameters or invalid query format.
401
Unauthorized
Missing or invalid API key in the X-API-Key header.
404
Not Found
No results found for the given search query.
429
Too Many Requests
Rate limit exceeded (100 req/min). Slow down and retry after 60 seconds.
500
Internal Server Error
Something went wrong on our end. Please try again later.

import requests

BASE_URL = "https://zentrix-api.onrender.com/v1"
API_KEY = "your_api_key_here"

def download_movie(title: str, quality: str = "1080") -> dict:
    resp = requests.get(
        f"{BASE_URL}/api/download",
        headers={"X-API-Key": API_KEY},
        params={"query": title, "type": "movie", "quality": quality}
    )
    resp.raise_for_status()
    return resp.json()

def download_episode(title: str, season: int, episode: int) -> dict:
    resp = requests.get(
        f"{BASE_URL}/api/download",
        headers={"X-API-Key": API_KEY},
        params={"query": title, "type": "series", "season": season, "episode": episode}
    )
    resp.raise_for_status()
    return resp.json()

# Usage
data = download_movie("Inception", quality="1080")
print(f"Title: {data['data']['title']}")
print(f"Download: {data['data']['download_url']}")
const ZENTRIX_BASE = 'https://zentrix-api.onrender.com/v1';
const API_KEY = 'your_api_key_here';

async function getDownloadLink({ query, type = 'movie', quality = '1080', season, episode } = {}) {
  const params = new URLSearchParams({ query, type, quality });
  if (season) params.append('season', season);
  if (episode) params.append('episode', episode);

  const res = await fetch(`${ZENTRIX_BASE}/api/download?${params}`, {
    headers: { 'X-API-Key': API_KEY, 'Accept': 'application/json' }
  });

  if (!res.ok) throw new Error(`HTTP ${res.status}`);
  return res.json();
}

// Movie
getDownloadLink({ query: 'Inception' })
  .then(({ data }) => console.log(data.download_url))
  .catch(console.error);

// Episode
getDownloadLink({ query: 'Breaking Bad', type: 'series', season: 1, episode: 1 })
  .then(({ data }) => console.log(data.download_url));
// Node.js with axios
const axios = require('axios');

const zentrix = axios.create({
  baseURL: 'https://zentrix-api.onrender.com/v1',
  headers: { 'X-API-Key': process.env.ZENTRIX_API_KEY }
});

async function search(query, options = {}) {
  const { data } = await zentrix.get('/api/download', {
    params: { query, ...options }
  });
  return data;
}

// Async/await usage
(async () => {
  try {
    const result = await search('Inception', { type: 'movie', quality: '1080' });
    console.log('Download URL:', result.data.download_url);
    console.log('Subtitle URL:', result.data.subtitle_url);
  } catch (err) {
    if (err.response?.status === 429) {
      console.error('Rate limited — wait 60s and retry');
    } else {
      console.error(err.message);
    }
  }
})();
